Chosen (Chosen, angular-chosen)

Chosen allows for searchable dropdowns.

Provide a chosen attribute with an ng-options attribute to a select tag to search for elements.

<select chosen
  ng-model="superhero"
  ng-options="s for s in superheroes">
</select>
<option value=""></option>
<label>You have chosen {{ superhero }}!</label>
            
(function(PatternLibrary) {
  'use strict';
  PatternLibrary.controller('ChosenDemoCtrl', ['$scope', function($scope) {
    $scope.superheroes = [
      'Batman',
      'Superman',
      'Green Arrow',
      'Wonder Woman'
    ];
  }]);
})(angular.module('PatternLibrary'));
