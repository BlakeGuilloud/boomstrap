Scores

Options

Color options for ranking / scoring data.


      <script type="text/javascript">
        angular.module('boomstrap').controller('btScoreTest', function($scope) {
          $scope.scores {
            excellent: 99,
            good: 75,
            average: 50,
            'default': 25
          };
        });
      </script>
      <bt-score score="scores.excellent"/>
      <bt-score score="scores.good"/>
      <bt-score score="scores.average"/>
      <bt-score score="scores.default"/>

Sizes

Fancy larger or smaller scores? Add size='lg', size='sm', or size='xs' to the directive for additional sizes.

      
    <p>
      <bt-score score="scores.excellent" size="lg"></bt-score>
      <bt-score score="scores.good" size="lg"></bt-score>
      <bt-score score="scores.average" size="lg"></bt-score>
      <bt-score score="scores.default" size="lg"></bt-score>
