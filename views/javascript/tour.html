// Instance the tour
var doctour = new Tour({
    name: "example",
    template: "<div class='popover tour'>" +
        "<div class='arrow'></div>" +
        "<div class='popover-close'><i class='ficon ficon-cross' data-role='end'></i></div>" +
        "<h3 class='popover-title'></h3>" +
        "<div class='popover-content'></div>" +
        "<div class='popover-navigation'><button class='btn btn-sm btn-default' data-role='prev'>Prev</button><button class='btn btn-sm btn-primary js-next' data-role='next'>Next</button></div>"+
        "</div>"
});
// Add your steps.
doctour.addSteps([
  {
      element: "#stop1",
      title: "Welcome to the Bootstrap Tour!",
      content: "Introduce new users to your product by walking them through it step by step.",
      placement: "right"
  },
  {
      element: "#stop2",
      title: "Title of my step",
      content: "<ul><li>Lorem ipsum</li><li>Dolor sit amet</li></ul>",
      placement: "top"
  },
  {
      element: "#stop3",
      title: "Title of my step",
      content: "Content of my step",
      placement: "right"
  }
  {
      element: "#stop4",
      title: "Title of my step",
      content: "Content of my step",
      placement: "top",
      onShown: function () {
          $('.js-next').attr('data-role', 'end');
          $('.js-next').removeClass('disabled');
          $('.js-next').text('Got It! ');
      }
  }
]);

// Initialize the tour
doctour.init();
